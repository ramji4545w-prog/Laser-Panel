# ---------- AUTH ----------
def login_required(fn):
    def wrap(*a, **k):
        if not session.get("user"):
            return redirect("/login")
        return fn(*a, **k)
    wrap.__name__ = fn.__name__
    return wrap

@app.route("/login", methods=["GET","POST"])
def login():
    err = ""
    if request.method=="POST":
        u = request.form.get("user"); p = request.form.get("pass")
        row = cur.execute("SELECT * FROM admins WHERE user=? AND pass=?", (u,p)).fetchone()
        if row:
            session["user"] = u
            session["role"] = row[2]
            return redirect("/dashboard")
        else:
            err = "Invalid credentials"
    return f"""
    <body style='background:#0b0b0b;color:#fff;font-family:Segoe UI;display:flex;align-items:center;justify-content:center;height:100vh'>
    <form method='post' style='background:#111;padding:20px;border-radius:10px'>
    <h2 style='color:#ff3b3b'>Login</h2>
    <input name='user' placeholder='Username'><br><br>
    <input name='pass' placeholder='Password' type='password'><br><br>
    <button style='background:#ff3b3b;color:#fff;border:none;padding:8px 12px'>Login</button>
    <div style='color:red'>{err}</div>
    </form></body>
    """

@app.route("/logout")
