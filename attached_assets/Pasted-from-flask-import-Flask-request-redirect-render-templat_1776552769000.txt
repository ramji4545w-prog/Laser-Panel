# ---------------- HOME ----------------
@app.route("/")
def home():
    conn = db()

    today = datetime.date.today()

    users = conn.execute("SELECT * FROM users WHERE status='pending'").fetchall()

    today_reg = conn.execute(
        "SELECT COUNT(*) FROM users WHERE date(created_at)=?",
        (today,)
    ).fetchone()[0]

    today_dep = conn.execute(
        "SELECT SUM(amount) FROM users WHERE status='accepted' AND date(created_at)=?",
        (today,)
    ).fetchone()[0] or 0

    total_users = conn.execute("SELECT COUNT(*) FROM users").fetchone()[0]

    total_dep = conn.execute(
        "SELECT SUM(amount) FROM users WHERE status='accepted'"
    ).fetchone()[0] or 0

    return render_template_string(
        HTML,
        users=users,
        today_reg=today_reg,
        today_dep=today_dep,
        total_users=total_users,
        total_dep=total_dep
    )
